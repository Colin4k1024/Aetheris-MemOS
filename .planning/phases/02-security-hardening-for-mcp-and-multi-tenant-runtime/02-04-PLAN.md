         ```rust
         use metrics::{increment_counter, counter};

         /// Record an isolation violation
         pub fn record_isolation_violation(
             tenant_id: &TenantId,
             attempted_resource: &str,
             reason: &str,
         ) {
             // Increment counter for monitoring
             increment_counter!(
                 "multi_tenant_isolation_violations_total",
                 "tenant_id" => tenant_id.as_str(),
                 "resource" => attempted_resource,
                 "reason" => reason
             );

             // Structured log for audit trail
             tracing::warn!(
                 tenant_id = %tenant_id,
                 resource = attempted_resource,
                 reason = reason,
                 "MULTI-TENANT ISOLATION VIOLATION DETECTED"
             );
         }
         ```

    2. Enhance `TenantIsolationLayer::verify_access()`:
       ```rust
       pub fn verify_access(&self, memory: &MemoryEntry, tenant_id: &TenantId) -> bool {
           if !self.config.enforce {
               return true;
           }
           let belongs = memory.source_id.starts_with(&tenant_id.prefix());
           if !belongs {
               record_isolation_violation(
                   tenant_id,
                   &memory.entry_id,
                   "cross_tenant_access_attempted"
               );
           }
           belongs
       }
       ```

       ```rust
           // First check if entry belongs to tenant

                   return Ok(None);  // Don't leak that entry exists
               }
           }

           Ok(entry)
       }
       ```
